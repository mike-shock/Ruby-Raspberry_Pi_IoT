def read_sensor
  # Read data back from MAX44009_REG_LUX_HIGH(0x03), 2 bytes, luminance MSB, luminance LSB
  data = @device.i2cget(MAX44009_REG_LUX_HIGH, 2)
  @value = data_to_lux(data)
end
