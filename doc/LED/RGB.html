def blink(list = :all, repeat=3, switched_on=0.25, switched_off=0.5)
  colors = (list == :all) ? [:red, :green, :blue] : (list.class == Array ? list : [list])
  repeat.times do |n|
    colors.each do |color|
      on color
      sleep switched_on
      off color
      sleep switched_off
    end
  end
end

def off(color=:all)
  color = :aoi if color == :aqua
  case color
  when :red
    red_off
  when :green
    green_off
  when :blue
    blue_off
  when :yellow
    red_off
    green_off
  when :aoi
    blue_off
    green_off
  when :all
    all_off
  else
    all_off
  end
end
