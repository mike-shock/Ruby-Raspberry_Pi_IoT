def wait_for_presses(attempts=2)
  @intervals = []
  attempts.times do
    interval = []
    timer = Time.now
    while not_pressed? do
      sleep @measure_pause
    end
    interval << Time.now-timer
    timer = Time.now
    while pressed? do
      sleep @measure_pause
    end
    interval << Time.now-timer
    @intervals << interval
  end
  @intervals
end
